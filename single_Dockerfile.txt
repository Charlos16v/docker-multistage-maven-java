# Linux with OpenJDK JRE
FROM openjdk:11.0-jre-slim-buster

LABEL "edu.elsmancs.gildedrose"="Kata Gilded Rose" \
        version="1.0" \
        description="Kata Gilded Rose en Java" \
        maintainer="davig@cifpfbmoll.eu"

# las rutas relativas de COPY y ADD se crean 
# a partir de este directorio (se crea)
WORKDIR $HOME/app

# copiar el jar desde el local al workdir del docker
COPY ./target/gildedrose-1.0-SNAPSHOT.jar ./gildedrose.jar

# ejecutar este comando al ejecutar el docker
CMD ["java", "-jar", "./gildedrose.jar"]
